================================================================================
DATA STRUCTURES
================================================================================

OWNER - Active Trades Format:
  trade = {
    id = "trade_001",
    offer = {itemType = "Base.Water", quantity = 1},
    request = {itemType = "Base.Nails", quantity = 100}
  }

CUSTOMER - Available Trades:
  Same format as above

SERVER COMMAND - Set Trades:
  {
    action = "SET_TRADES",
    itemType = "Base.Water",
    trades = {...}
  }

SERVER COMMAND - Buy Trade:
  {
    action = "BUY_TRADE",
    itemType = "Base.Water",
    tradeIndex = 1,
    offer = {itemType, quantity},
    request = {itemType, quantity}
  }

COMPONENT USAGE
================================================================================

ESC_QuantityInput:
  Usage: self.input = ESC_QuantityInput:new(x, y, w, h, "1", maxVal)
  Methods: getValue(), setValue(), setMaxValue(), isValidValue()
  Callbacks: .onQuantityChanged

ESC_ItemInput:
  Usage: self.input = ESC_ItemInput:new(x, y, w, h, "Base.Nails")
  Methods: getValue(), setValue(), isValidValue(), getItemScript()
  Callbacks: .onItemChanged

ESC_TradeCard:
  Usage: self.card = ESC_TradeCard:new(x, y, w, h, trade)
  Methods: setTrade(), getTrade(), setSelected(), setShowArrow()
  Callbacks: .onTradeCardClick

ESC_TradeSelector:
  Usage: self.sel = ESC_TradeSelector:new(x, y, w, h, trades)
  Methods: setTrades(), selectTrade(), getSelectedTrade(), getTradeCount()
  Callbacks: .onTradeSelected

ESC_AffordabilityPanel:
  Usage: self.panel = ESC_AffordabilityPanel:new(x, y, w, h)
  Methods: setNeeds(), setPlayerInventory(), setMaxTrades(), canAffordTrade()
